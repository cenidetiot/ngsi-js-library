Entities Functions

Parse an Entity

    var entity = ngsi.parseEntity({
        id :'Room1',
        type:'Room',
        temperature : {
            value : 50 ,
            metadata : {
                frecuency: 40,
                scale: 'Celsious'
            }
        },
        dateStamp : new Date()
    })

Output

    {
        "id":"Room1",
        "type":"Room",
        "temperature":{
            "value":50,
            "type":"Number",
            "metadata":{
                "frecuency":{
                    "value":40,
                    "type":"Number"
                },
                "scale":{
                    "value":"Celsious",
                    "type":"Text"
                }
            }
        },
        "dateStamp":{
            "value":"2017-10-08T04:01:19.560Z",
            "type":"DateTime",
            "metadata":{}
        }
    }

Parse an Attribute

    var attribute = ngsi.parseAttrs({
        temperature : {
            value : 50,
            metadata :{
                frecuency : 50,
                scale: 'Fahrenheit'
            }
        }
    })

Output

    {
        "temperature":{
            "value":50,
            "type":"Number",
            "metadata":{
                "frecuency":{
                    "value":50,
                    "type":"Number"
                },
                "scale":{
                    "value":"Fahrenheit",
                    "type":"Text"
                }
            }
        }
    }

Parse a Value

    var value = ngsi.parseValue(50)

Output

    {
         "value": 50,
         "type": "Number",
         "metadata": {}
     }
